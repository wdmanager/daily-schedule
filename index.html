Slack通知設定
GASスクリプトをコピーしてGoogleスプレッドシートに貼り付けてください
通知の仕組み
毎日 22:00 に未入力メンバーへメールを自動送信。
その日が休み・婚礼撮影日に登録されているメンバーはスキップ。
マネージャーには 22:35 に日次サマリー DM が届きます。
設定手順
1Slack App を作成し Bot Token (xoxb-...) を取得
2各メンバーの Slack User ID を確認（プロフィール → 「コピー」）
3下記スクリプトをコピー → スプレッドシートの「拡張機能 → Apps Script」に貼り付け
4SLACK_TOKEN と MEMBERS の slackId を書き換える
5「トリガーを追加」→ 関数: sendDailyReminder → 時間主導型 → 毎日 22:00〜23:00
GASスクリプト
// ========================================
// Kuppography Slack通知スクリプト
// Googleスプレッドシートの Apps Script に貼り付けて使用
// ========================================

// ▼ ここを自分の情報に書き換えてください
const SLACK_TOKEN = 'xoxb-YOUR-TOKEN-HERE';
const MANAGER_SLACK_ID = 'U0XXXXXXXXX'; // マネージャーのSlack User ID
const APP_URL = 'https://your-app-url-here'; // アプリのURL

const MEMBERS = [
  { name: '小林 俊介',  slackId: 'U0XXXXXXXXX' },
  { name: '飯倉 孝史',   slackId: 'U0XXXXXXXXX' },
  { name: '佐々木 大翔', slackId: 'U0XXXXXXXXX' },
  { name: '川上 輝',   slackId: 'U0XXXXXXXXX' },
  { name: '坂本 楓花',   slackId: 'U0XXXXXXXXX' },
  { name: '山口 葵',   slackId: 'U0XXXXXXXXX' },
  { name: '横山 大燿',   slackId: 'U0XXXXXXXXX' },
  { name: '片倉 来波',   slackId: 'U0XXXXXXXXX' },
];

// ========================================
// メイン関数（毎日22:00に自動実行）
// ========================================
function sendDailyReminder() {
  const today = getDateStr(new Date());
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const reportSheet = ss.getSheetByName('日次データ');
  const offSheet = ss.getSheetByName('休み・婚礼登録');

  const unsubmitted = [];

  MEMBERS.forEach(member => {
    // 休み・婚礼日チェック → 入力不要日はスキップ
    if (checkDayOff(offSheet, member.name, today)) return;

    // 終業報告の入力確認
    if (!checkSubmitted(reportSheet, member.name, today, 'report')) {
      unsubmitted.push(member);
      sendSlackDM(member.slackId,
        `:alarm_clock: *終業報告がまだです*\n` +
        `今日の報告を入力してください。\n` +
        `${APP_URL}`
      );
    }
  });

  // マネージャーにサマリー送信（22:35頃）
  Utilities.sleep(5 * 60 * 1000); // 5分待機
  const total = MEMBERS.length;
  const submitted = total - unsubmitted.length;
  let msg = `:bar_chart: *本日の終業報告サマリー* (${today})\n`;
  msg += `完了: ${submitted}/${total}名\n`;
  if (unsubmitted.length > 0) {
    msg += `:x: 未提出: ${unsubmitted.map(m => m.name).join('、')}`;
  } else {
    msg += `:white_check_mark: 全員提出済みです`;
  }
  sendSlackDM(MANAGER_SLACK_ID, msg);
}

// ========================================
// ユーティリティ関数
// ========================================
function sendSlackDM(userId, text) {
  UrlFetchApp.fetch('https://slack.com/api/chat.postMessage', {
    method: 'post',
    contentType: 'application/json',
    headers: { 'Authorization': 'Bearer ' + SLACK_TOKEN },
    payload: JSON.stringify({ channel: userId, text: text, unfurl_links: false }),
    muteHttpExceptions: true
  });
}

function checkDayOff(sheet, name, dateStr) {
  const data = sheet.getDataRange().getValues();
  for (let i = 1; i < data.length; i++) {
    if (data[i][0] === name && data[i][1] === dateStr) return true;
  }
  return false;
}

function checkSubmitted(sheet, name, dateStr, type) {
  const data = sheet.getDataRange().getValues();
  for (let i = 1; i < data.length; i++) {
    if (data[i][0] === name && data[i][1] === dateStr && data[i][2] === type) return true;
  }
  return false;
}

function getDateStr(date) {
  return date.getFullYear() + '-' +
    String(date.getMonth() + 1).padStart(2, '0') + '-' +
    String(date.getDate()).padStart(2, '0');
}
